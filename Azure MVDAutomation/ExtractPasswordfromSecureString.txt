#Purpose: Pulls the password out of a SecureString\



$cred = Get-Credential

#Example
#$cred
#UserName                     Password
#--------                     --------
#test     System.Security.SecureString

$test = $cred.Password


$pass = [System.Runtime.InteropServices.marshal]::PtrToStringAuto([System.Runtime.InteropServices.marshal]::SecureStringToBSTR($test))
$pass
